**3.1 Immutable Timestamp Protocol** [R]

**Purpose:** Create legally defensible, tamper-proof timestamp records

**Implementation:**

```python
import hashlib
import hmac
from datetime import datetime
import json

class CryptographicTimestampSigner:
    """
    Generate court-admissible cryptographic timestamp signatures
    Meets Federal Rules of Evidence 901-902 requirements
    """
    
    def __init__(self, secret_key):
        """
        Initialize with secret key (stored in HSM for high-stakes deployments)
        
        Args:
            secret_key: Cryptographic secret for HMAC signing
        """
        self.secret_key = secret_key
    
    def sign_timestamp(self, event_data):
        """
        Create cryptographically signed timestamp entry
        
        Args:
            event_data: Dict containing:
                - event_type: "input" | "output" | "processing"
                - timestamp: ISO 8601 with milliseconds
                - content_hash: SHA-256 of actual content
                - metadata: Additional context
        
        Returns:
            Signed timestamp entry with cryptographic proof
        """
        # Canonical representation
        canonical = json.dumps(event_data, sort_keys=True)
        
        # Generate signature using HMAC-SHA256
        signature = hmac.new(
            self.secret_key.encode(),
            canonical.encode(),
            hashlib.sha256
        ).hexdigest()
        
        # Create signed entry
        signed_entry = {
            'event_data': event_data,
            'signature': signature,
            'signature_algorithm': 'HMAC-SHA256',
            'signed_at': datetime.utcnow().isoformat() + 'Z',
            'signer_version': 'temporal-protocol-v1.6'
        }
        
        return signed_entry
    
    def verify_signature(self, signed_entry, expected_secret_key):
        """
        Verify cryptographic signature of timestamp entry
        
        Returns:
            Boolean: True if signature valid, False if tampered
        """
        event_data = signed_entry['event_data']
        claimed_signature = signed_entry['signature']
        
        # Recompute signature
        canonical = json.dumps(event_data, sort_keys=True)
        computed_signature = hmac.new(
            expected_secret_key.encode(),
            canonical.encode(),
            hashlib.sha256
        ).hexdigest()
        
        # Constant-time comparison (prevents timing attacks)
        return hmac.compare_digest(claimed_signature, computed_signature)
    
    def create_chain_of_custody(self, tc_entry):
        """
        Create full chain of custody for temporal calibration entry
        
        Returns:
            Chain of custody record with all timestamps signed
        """
        chain = {
            'tc_id': tc_entry['tc_id'],
            'custody_chain': []
        }
        
        # Sign T0 (previous output)
        t0_signed = self.sign_timestamp({
            'event_type': 'previous_output',
            'timestamp': tc_entry['measurement']['t0_previous_output'],
            'content_hash': self.hash_content(tc_entry.get('previous_content', ''))
        })
        chain['custody_chain'].append(t0_signed)
        
        # Sign T1 (current input)
        t1_signed = self.sign_timestamp({
            'event_type': 'current_input',
            'timestamp': tc_entry['measurement']['t1_current_input'],
            'content_hash': self.hash_content(tc_entry.get('input_content', '')),
            'task_category': tc_entry['task']['category']
        })
        chain['custody_chain'].append(t1_signed)
        
        # Sign T2 (processing complete)
        t2_signed = self.sign_timestamp({
            'event_type': 'processing_complete',
            'timestamp': tc_entry['measurement']['t2_processing_complete'],
            'content_hash': self.hash_content(tc_entry.get('output_content', '')),
            'processing_latency_seconds': tc_entry['measurement']['processing_latency_seconds']
        })
        chain['custody_chain'].append(t2_signed)
        
        # Sign entire chain
        chain_signature = self.sign_timestamp({
            'event_type': 'chain_complete',
            'timestamp': datetime.utcnow().isoformat() + 'Z',
            'chain_hash': self.hash_content(json.dumps(chain, sort_keys=True)),
            'custody_entries': len(chain['custody_chain'])
        })
        
        chain['chain_signature'] = chain_signature
        chain['chain_integrity'] = 'VERIFIED'
        
        return chain
    
    def hash_content(self, content):
        """Generate SHA-256 hash of content"""
        return hashlib.sha256(content.encode()).hexdigest()
```

**For highest-stakes deployments (medical, legal, safety-critical):**

```python
class ImmutableAuditTrail:
    """
    Append-only audit trail with Merkle tree verification
    Meets SOC 2, HIPAA, ISO 27001 audit requirements
    """
    
    def __init__(self):
        self.entries = []
        self.merkle_roots = []
    
    def append_entry(self, tc_entry, signed_chain):
        """
        Append timestamp entry to immutable trail
        
        Args:
            tc_entry: Temporal calibration entry
            signed_chain: Cryptographically signed chain of custody
        """
        entry = {
            'index': len(self.entries),
            'timestamp': datetime.utcnow().isoformat() + 'Z',
            'tc_entry': tc_entry,
            'signed_chain': signed_chain,
            'previous_hash': self.get_latest_hash()
        }
        
        # Calculate entry hash
        entry['entry_hash'] = self.hash_entry(entry)
        
        # Append to trail
        self.entries.append(entry)
        
        # Update Merkle root every 10 entries
        if len(self.entries) % 10 == 0:
            merkle_root = self.calculate_merkle_root()
            self.merkle_roots.append({
                'root': merkle_root,
                'timestamp': entry['timestamp'],
                'entries_count': len(self.entries)
            })
        
        return entry['entry_hash']
    
    def verify_integrity(self):
        """
        Verify entire audit trail has not been tampered with
        
        Returns:
            Boolean: True if chain intact, False if compromised
        """
        for i in range(1, len(self.entries)):
            current = self.entries[i]
            expected_previous = self.entries[i-1]['entry_hash']
            
            if current['previous_hash'] != expected_previous:
                return {
                    'integrity': False,
                    'tampered_at_index': i,
                    'reason': 'Hash chain broken'
                }
        
        return {
            'integrity': True,
            'entries_verified': len(self.entries),
            'merkle_roots': len(self.merkle_roots)
        }
    
    def hash_entry(self, entry):
        """Generate SHA-256 hash of entry"""
        # Create canonical representation
        canonical = json.dumps({
            'index': entry['index'],
            'timestamp': entry['timestamp'],
            'tc_id': entry['tc_entry']['tc_id'],
            'previous_hash': entry['previous_hash']
        }, sort_keys=True)
        
        return hashlib.sha256(canonical.encode()).hexdigest()
    
    def get_latest_hash(self):
        """Get hash of most recent entry"""
        if not self.entries:
            return '0' * 64  # Genesis hash
        return self.entries[-1]['entry_hash']
    
    def calculate_merkle_root(self):
        """Calculate Merkle root for current entries"""
        if not self.entries:
            return None
        
        # Simplified Merkle tree (production would use full binary tree)
        hashes = [e['entry_hash'] for e in self.entries]
        
        while len(hashes) > 1:
            if len(hashes) % 2 == 1:
                hashes.append(hashes[-1])  # Duplicate last if odd
            
            new_hashes = []
            for i in range(0, len(hashes), 2):
                combined = hashes[i] + hashes[i+1]
                new_hash = hashlib.sha256(combined.encode()).hexdigest()
                new_hashes.append(new_hash)
            
            hashes = new_hashes
        
        return hashes[0]
    
    def export_for_legal_discovery(self, tc_id=None):
        """
        Export audit trail in format suitable for legal discovery
        Meets EDRM (Electronic Discovery Reference Model) standards
        """
        export = {
            'export_timestamp': datetime.utcnow().isoformat() + 'Z',
            'export_version': 'temporal-protocol-v1.6',
            'total_entries': len(self.entries),
            'integrity_verified': self.verify_integrity()['integrity'],
            'entries': []
        }
        
        for entry in self.entries:
            if tc_id is None or entry['tc_entry']['tc_id'] == tc_id:
                export['entries'].append({
                    'index': entry['index'],
                    'timestamp': entry['timestamp'],
                    'tc_id': entry['tc_entry']['tc_id'],
                    'entry_hash': entry['entry_hash'],
                    'signed_chain': entry['signed_chain'],
                    'tc_entry': entry['tc_entry']
                })
        
        return export
```
